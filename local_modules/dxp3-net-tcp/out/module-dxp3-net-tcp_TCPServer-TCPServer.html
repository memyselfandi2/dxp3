Example
let tcp = require('dxp3-net-tcp');
// Set our host, port and timeout
let args = {};
args.address = '127.0.0.1';
args.port = 58761;
args.timeout = 10000;
// Create a new TCPServer
let myTCPServer = new tcp.TCPServer(args);
// Start the server
myTCPServer.start();
